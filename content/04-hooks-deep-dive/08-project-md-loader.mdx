## 目标:让 .md 文件像组件一样 import

```tsx filename="src/App.tsx"
import Post from "./posts/hello.md"
import Guide from "./docs/guide.md"

export function App() {
  return (
    <div>
      <Post /> {/* 直接当 React 组件用 */}
      <Guide />
    </div>
  )
}
```

这是很多内容驱动应用的需求:内容写在 `.md` 文件里,代码里直接 import 和渲染。

## 插件设计

整体流程:

```text
import Post from "./posts/hello.md"
              │
              ▼
resolveId("./posts/hello.md") → 解析为文件绝对路径
              │
              ▼
load("/abs/posts/hello.md")   → 读取 .md 文件内容
              │
              ▼
transform(markdownCode, id)   → 把 Markdown 转换为 React 组件代码
              │
              ▼
浏览器收到的是:
  import { jsx as _jsx } from "react/jsx-runtime"
  export default function MarkdownComponent() {
    return _jsx("div", { dangerouslySetInnerHTML: { __html: "..." } })
  }
```

## 步骤一:最小骨架

```ts filename="src/plugin.ts"
import type { Plugin } from "vite"
import { readFileSync } from "node:fs"

export function mdLoader(): Plugin {
  return {
    name: "vite-plugin-md",

    // 只拦截 .md 文件的 load
    load(id) {
      if (!id.endsWith(".md")) return null

      const content = readFileSync(id, "utf-8")
      // 先返回原始 markdown 字符串,步骤二再转换
      return `export default ${JSON.stringify(content)}`
    },
  }
}
```

此时 `import Post from "./hello.md"` 会给你一个字符串,而不是组件。

## 步骤二:把 Markdown 转为 HTML

安装 `marked`(Markdown 解析器,已在 package.json 里):

```ts filename="src/plugin.ts"
import type { Plugin } from "vite"
import { readFileSync } from "node:fs"
import { marked } from "marked"

export function mdLoader(): Plugin {
  return {
    name: "vite-plugin-md",

    load(id) {
      if (!id.endsWith(".md")) return null

      const markdown = readFileSync(id, "utf-8")
      // 把 Markdown 转换为 HTML 字符串
      const html = marked(markdown) as string

      // 临时:直接返回 HTML 字符串(还不是组件)
      return `export default ${JSON.stringify(html)}`
    },
  }
}
```

## 步骤三:包装成 React 组件

把 HTML 字符串包装成 React 组件:

```ts filename="src/plugin.ts"
import type { Plugin } from "vite"
import { readFileSync } from "node:fs"
import { marked } from "marked"

export function mdLoader(): Plugin {
  return {
    name: "vite-plugin-md",

    load(id) {
      if (!id.endsWith(".md")) return null

      const markdown = readFileSync(id, "utf-8")
      const html = marked(markdown) as string

      // 生成合法的 React 组件代码
      return `
import { createElement } from "react"

export default function MarkdownComponent() {
  return createElement("div", {
    className: "markdown-body",
    dangerouslySetInnerHTML: { __html: ${JSON.stringify(html)} },
  })
}
`
    },
  }
}
```

在 `vite.config.ts` 里使用:

```ts filename="vite.config.ts"
import { defineConfig } from "vite"
import react from "@vitejs/plugin-react"
import { mdLoader } from "./src/plugin"

export default defineConfig({
  plugins: [react(), mdLoader()],
})
```

## 步骤四:添加 TypeScript 类型声明

直接 `import Post from "./hello.md"` 会让 TypeScript 报错。需要声明模块类型:

```ts filename="src/md.d.ts"
declare module "*.md" {
  import type { ComponentType } from "react"
  const Component: ComponentType
  export default Component
}
```

把这个文件加入 `tsconfig.json` 的 `include` 路径即可。

## 步骤五:添加 HMR 支持

修改 `.md` 文件后,如果没有 HMR,需要手动刷新页面。实现自动热更新:

```ts filename="src/plugin.ts"
import type { Plugin } from "vite"
import { readFileSync } from "node:fs"
import { marked } from "marked"

export function mdLoader(): Plugin {
  return {
    name: "vite-plugin-md",

    load(id) {
      if (!id.endsWith(".md")) return null

      const markdown = readFileSync(id, "utf-8")
      const html = marked(markdown) as string

      return `
import { createElement } from "react"

export default function MarkdownComponent() {
  return createElement("div", {
    className: "markdown-body",
    dangerouslySetInnerHTML: { __html: ${JSON.stringify(html)} },
  })
}
`
    },

    // 让 .md 文件支持 HMR
    handleHotUpdate({ file, modules, server }) {
      if (!file.endsWith(".md")) return

      // Vite 已经自动把受影响的 ModuleNode 放在 modules 里
      // 直接返回即可触发这些模块的 HMR
      // (Vite 会发送 { type: "update" } 消息给浏览器)
      return modules
    },
  }
}
```

## 进阶:Frontmatter 支持

很多 .md 文件有 frontmatter(YAML 头部):

```markdown filename="hello.md"
---
title: 你好世界
date: 2026-06-26
tags: [vite, plugin]
---

# 正文内容

...
```

扩展插件支持导出 frontmatter:

```ts filename="src/plugin.ts"
import type { Plugin } from "vite"
import { readFileSync } from "node:fs"
import { marked } from "marked"

function parseFrontmatter(content: string): { data: Record<string, unknown>; body: string } {
  const match = content.match(/^---\r?\n([\s\S]*?)\r?\n---\r?\n([\s\S]*)$/)
  if (!match) return { data: {}, body: content }

  // 简单 YAML 解析(仅处理 key: value 格式)
  const data: Record<string, unknown> = {}
  for (const line of match[1].split("\n")) {
    const colonIdx = line.indexOf(":")
    if (colonIdx === -1) continue
    const key = line.slice(0, colonIdx).trim()
    const val = line
      .slice(colonIdx + 1)
      .trim()
      .replace(/^["']|["']$/g, "")
    data[key] = val
  }

  return { data, body: match[2] }
}

export function mdLoader(): Plugin {
  return {
    name: "vite-plugin-md",
    load(id) {
      if (!id.endsWith(".md")) return null

      const content = readFileSync(id, "utf-8")
      const { data: frontmatter, body } = parseFrontmatter(content)
      const html = marked(body) as string

      return `
import { createElement } from "react"

export const frontmatter = ${JSON.stringify(frontmatter)}

export default function MarkdownComponent() {
  return createElement("div", {
    className: "markdown-body",
    dangerouslySetInnerHTML: { __html: ${JSON.stringify(html)} },
  })
}
`
    },
    handleHotUpdate({ file, modules }) {
      if (!file.endsWith(".md")) return
      return modules
    },
  }
}
```

使用:

```tsx
import Post, { frontmatter } from "./posts/hello.md"

console.log(frontmatter.title)  // "你好世界"
<Post />
```
