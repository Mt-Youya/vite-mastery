## Server Entry

```tsx filename="src/entry-server.tsx"
import { renderToString } from "react-dom/server"
import { App } from "./App"

export async function render(url: string): Promise<{ html: string; data: unknown }> {
  // Server-side data fetching happens here
  const data = {
    message: `Data fetched on the server (${new Date().toISOString()})`,
  }

  const html = renderToString(<App url={url} data={data} />)

  return { html, data }
}
```

## Client Entry

```tsx filename="src/main.tsx"
import { hydrateRoot } from "react-dom/client"
import { App } from "./App"

// Read the server-injected data from window.__SSR_DATA__
const data = (window as { __SSR_DATA__?: unknown }).__SSR_DATA__

hydrateRoot(document.getElementById("root")!, <App url={window.location.pathname} data={data as { message: string }} />)
```

## Development Server

```js filename="server.js"
import { createServer as createViteServer } from "vite"
import { createServer as createHttpServer } from "node:http"
import { readFileSync } from "node:fs"

const isProd = process.argv.includes("--prod")

async function startServer() {
  let vite
  let template

  if (!isProd) {
    // Dev mode: start the Vite dev server
    vite = await createViteServer({
      server: { middlewareMode: true },
      appType: "custom",
    })
  } else {
    // Production mode: read the pre-built output
    template = readFileSync("dist/client/index.html", "utf-8")
  }

  const server = createHttpServer(async (req, res) => {
    const url = req.url ?? "/"

    // Dev mode: let Vite handle static assets
    if (!isProd && vite) {
      // Check if this is a Vite request (starting with /@vite/, etc.)
      await new Promise((resolve) => {
        vite.middlewares(req, res, resolve)
      })
      if (res.writableEnded) return
    }

    try {
      let html: string
      let appHtml: string
      let data: unknown

      if (!isProd && vite) {
        // Dev mode: use ssrLoadModule to execute the server entry
        const { render } = await vite.ssrLoadModule("/src/entry-server.tsx")
        ;({ html: appHtml, data } = await render(url))

        const rawTemplate = readFileSync("index.html", "utf-8")
        html = await vite.transformIndexHtml(url, rawTemplate)
      } else {
        // Production mode: directly import the pre-built output
        const { render } = await import("./dist/server/entry-server.js")
        ;({ html: appHtml, data } = await render(url))
        html = template!
      }

      // Inject the rendered result
      html = html
        .replace("<!--ssr-outlet-->", appHtml)
        .replace("<!--ssr-data-->", JSON.stringify(data))

      res.setHeader("Content-Type", "text/html; charset=utf-8")
      res.end(html)
    } catch (e) {
      if (vite) vite.ssrFixStacktrace(e)
      console.error(e)
      res.statusCode = 500
      res.end(String(e))
    }
  })

  const port = 3000
  server.listen(port, () => {
    console.log(`SSR server: http://localhost:${port}`)
    if (!isProd) console.log("  Dev mode: HMR enabled")
  })
}

startServer()
```

## Running the Project

```bash
cd examples/ssr-with-env-api
pnpm install

# Dev mode (with HMR)
pnpm dev

# Production mode
pnpm build   # build client + server
node server.js --prod
```

Visit `http://localhost:3000` and view the page source — you should see complete HTML (not an empty div), which confirms that SSR is working correctly.
