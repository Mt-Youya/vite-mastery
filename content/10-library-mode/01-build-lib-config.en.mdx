## Library mode vs. application mode

| Dimension      | Application mode             | Library mode                  |
| -------------- | ---------------------------- | ----------------------------- |
| Entry point    | HTML file                    | TypeScript/JS file            |
| Output purpose | Deployed directly to browser | Imported by other projects    |
| Output formats | ESM (modern browsers)        | ESM + CJS + UMD               |
| External deps  | Bundled into output          | Typically as peerDependencies |
| index.html     | Required                     | Not required                  |

## Minimal library config

```ts filename="vite.config.ts"
import { defineConfig } from "vite"
import { resolve } from "node:path"

export default defineConfig({
  build: {
    lib: {
      // Entry file: the library's public API is exported from here
      entry: resolve("src/index.ts"),

      // Global variable name for UMD format
      name: "MyLibrary",

      // Output filename pattern
      fileName: "my-library",
      // Default outputs:
      //   my-library.js (ES Module)
      //   my-library.umd.js (UMD)
    },

    // Don't bundle these dependencies (consumers install them)
    rollupOptions: {
      external: ["react", "react-dom"],
      output: {
        globals: {
          react: "React",
          "react-dom": "ReactDOM",
        },
      },
    },
  },
})
```

## `build.lib` configuration in detail

### `entry` — entry file

```ts
lib: {
  // Single entry
  entry: "src/index.ts",

  // Multiple entries (each entry generates a separate file)
  entry: {
    index: "src/index.ts",
    utils: "src/utils/index.ts",
    components: "src/components/index.ts",
  },
},
```

### `formats` — output formats

```ts
lib: {
  formats: ["es", "cjs"],  // Default: ["es", "umd"]
  // "es"   → ES Module (.js)
  // "cjs"  → CommonJS (.cjs)
  // "umd"  → UMD (requires name)
  // "iife" → Immediately Invoked Function Expression (browser global)
},
```

**Recommended combinations**:

- Modern npm packages: use only `["es", "cjs"]`
- Needs CDN usage: add `"umd"` or `"iife"`

### `fileName` — output filename

```ts
lib: {
  // String format: filename template
  fileName: "my-lib",
  // es:  my-lib.js
  // cjs: my-lib.cjs

  // Function format: fully customized
  fileName: (format, entryName) => `${entryName}.${format === "es" ? "js" : "cjs"}`,
},
```

## The companion package.json

Library mode requires a correctly configured `exports` field in `package.json`:

```json filename="package.json"
{
  "name": "@myorg/my-lib",
  "version": "1.0.0",
  "type": "module",

  "exports": {
    ".": {
      "import": "./dist/index.js", // ESM
      "require": "./dist/index.cjs", // CJS
      "types": "./dist/index.d.ts" // TypeScript types
    },
    "./utils": {
      "import": "./dist/utils.js",
      "require": "./dist/utils.cjs",
      "types": "./dist/utils.d.ts"
    }
  },

  "main": "./dist/index.cjs", // Legacy compatibility
  "module": "./dist/index.js", // Legacy ESM
  "types": "./dist/index.d.ts", // TypeScript

  "files": ["dist"],

  "peerDependencies": {
    "react": ">=18"
  }
}
```

## Complete example: React component library config

```ts filename="vite.config.ts"
import { defineConfig } from "vite"
import react from "@vitejs/plugin-react"
import { resolve } from "node:path"

export default defineConfig({
  plugins: [react()],
  build: {
    lib: {
      entry: resolve("src/index.ts"),
      name: "MyComponentLib",
      formats: ["es", "cjs"],
      fileName: (format) => `index.${format === "es" ? "js" : "cjs"}`,
    },
    rollupOptions: {
      // Don't bundle react — let consumers provide it
      external: ["react", "react/jsx-runtime", "react-dom"],
      output: {
        globals: {
          react: "React",
          "react/jsx-runtime": "jsxRuntime",
          "react-dom": "ReactDOM",
        },
        // Prevent CSS from being inlined into JS
        assetFileNames: "assets/[name][extname]",
      },
    },
  },
})
```
