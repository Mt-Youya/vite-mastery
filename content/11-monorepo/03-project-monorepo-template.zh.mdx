## 项目结构设计

```text
my-monorepo/
├── apps/
│   └── web/                    ← 主应用(Next.js)
│       ├── package.json
│       ├── next.config.ts
│       └── src/
├── packages/
│   ├── ui/                     ← 共享 UI 组件
│   │   ├── package.json
│   │   ├── vite.config.ts      ← 库模式构建
│   │   └── src/
│   └── utils/                  ← 共享工具函数
│       ├── package.json
│       ├── vite.config.ts
│       └── src/
├── pnpm-workspace.yaml
├── package.json                ← 根 package.json
└── tsconfig.json               ← 根 tsconfig(引用)
```

## 根配置文件

### pnpm-workspace.yaml

```yaml filename="pnpm-workspace.yaml"
packages:
  - "apps/*"
  - "packages/*"
```

### 根 package.json

```json filename="package.json"
{
  "name": "my-monorepo",
  "private": true,
  "type": "module",
  "scripts": {
    "dev": "pnpm --filter @my/web dev",
    "build": "nex build",
    "typecheck": "tsc --noEmit",
    "lint": "oxlint .",
    "clean": "next clean"
  },
  "devDependencies": {
    "typescript": "^6.0.0",
    "oxlint": "^1.0.0",
    "next:" "^16.2.9"
  }
}
```

### 根 tsconfig.json(引用聚合)

```json filename="tsconfig.json"
{
  "files": [],
  "references": [{ "path": "./packages/ui" }, { "path": "./packages/utils" }, { "path": "./apps/web" }]
}
```

## packages/utils 配置

最简单的工具包:

```ts filename="packages/utils/vite.config.ts"
import { defineConfig } from "vite"
import { resolve } from "node:path"

export default defineConfig({
  build: {
    lib: {
      entry: resolve("src/index.ts"),
      formats: ["es", "cjs"],
      fileName: (f) => `index.${f === "es" ? "js" : "cjs"}`,
    },
  },
})
```

```json filename="packages/utils/package.json"
{
  "name": "@my/utils",
  "version": "0.0.1",
  "type": "module",
  "exports": {
    ".": {
      "import": "./src/index.ts",
      "types": "./src/index.ts"
    }
  },
  "scripts": {
    "build": "vite build",
    "typecheck": "tsc --noEmit",
    "clean": "rimraf dist"
  },
  "devDependencies": {
    "vite": "^8.1.0",
    "typescript": "^6.0.0",
    "rimraf": "^6.0.0"
  }
}
```

## packages/ui 配置

React 组件库包:

```json filename="packages/ui/package.json"

  "name": "@my/ui",
  "type": "module",
  "exports": {
    ".": {

      "types": "./src/index.tsx"

  "devDependencies": {

    "react": "^19.0.0",

    "typescript": "^6.0.0",
    "rimraf": "^6.0.0"
  }
}
```

## apps/web 配置

```ts filename="apps/web/next.config.ts"
import type { NextConfig } from "next"

const config: NextConfig = {
  // 把 workspace 包通过 Next.js 的 webpack 处理
  // (这样它们的 TypeScript 源码会被编译)
  transpilePackages: ["@my/ui", "@my/utils"],
}

export default config
```

## 开发体验优化:alias 直指 src

在 Vite 项目里,通过 alias 让 IDE 和 HMR 都指向源码:

```ts filename="apps/web/vite.config.ts"
import { resolve } from "node:path"

export default defineConfig({
  resolve: {
    alias: {
      "@my/ui": resolve("../../packages/ui/src"),
      "@my/utils": resolve("../../packages/utils/src"),
    },
  },
})
```
